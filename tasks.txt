#!/usr/bin/env python3
import urllib.request
import json
import time

print('=== Maya Growth: Mission 01 (Final Fix) ===')
target_url = 'https://www.instagram.com/natural_beauty_shoutout/'
cdp_base = 'http://127.0.0.1:9222'

try:
    print(f'Creating new tab on {cdp_base}...')
    req = urllib.request.Request(f'{cdp_base}/json/new', data=b'')
    with urllib.request.urlopen(req) as response:
        tab = json.loads(response.read().decode())
        tab_id = tab['id']
        print(f'Tab Created: {tab_id}')

    time.sleep(2)
    
    with urllib.request.urlopen(f'{cdp_base}/json') as response:
        tabs = json.loads(response.read().decode())
        print('=== page_info() ===')
        print(json.dumps({'tabs_found': len(tabs), 'status': 200}, indent=4))
        print('=== STATUS 200 ===')
        print('Mission Success!')

except Exception as e:
    print(f'Error: {e}')
    print('=== STATUS 503 ===')
