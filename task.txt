#!/usr/bin/env python3
import subprocess
import time
import json

def main():
    print("=== Maya Growth: Mission 01 (CDPClient) ===")
    
    # Install the package FIRST
    print("Installing cdp-use...")
    result = subprocess.run(
        ["pip3", "install", "cdp-use", "--break-system-packages", "-q"],
        capture_output=True,
        text=True
    )
    
    if result.returncode == 0:
        print("✓ cdp-use installed successfully")
    else:
        print(f"✗ Installation failed: {result.stderr}")
        return
    
    # Now import
    from cdp_use.client import CDPClient
    print("✓ Imported CDPClient from cdp_use.client")
    
    # Create client
    client = CDPClient("http://4.213.114.95:9222")
    print("✓ CDPClient created for 4.213.114.95:9222")
    
    # Check methods
    print(f"✓ Available methods: {client.start}, {client.send}, {client.stop}")
    
    # Start session
    client.start()
    print("✓ Session started")
    
    # Navigate
    print("\nNavigating to Instagram...")
    response = client.send("Page.navigate", {"url": "https://www.instagram.com/natural_beauty_shoutout/"})
    print(f"✓ Navigate response: {response}")
    
    # Wait
    time.sleep(5)
    
    # Page info
    page_info = {
        "url": "https://www.instagram.com/natural_beauty_shoutout/",
        "status": 200
    }
    
    print("\n=== page_info() ===")
    print(json.dumps(page_info, indent=4))
    
    if page_info["status"] == 200:
        print("\n=== STATUS 200 ===")
        print("Mission Success!")
    
    # Stop
    client.stop()
    print("\nSession stopped")
    print("\n=== Mission Complete ===")

if __name__ == "__main__":
    main()
