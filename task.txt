#!/usr/bin/env python3
"""
Cloud-Strike: Shadow Backup & Cloud Leak Detection Tool
==========================================================
A script to search for cloud exposures and shadow backups associated with phone numbers.
"""

import phonenumbers
import urllib.parse
import requests
import time
import re
import json

phone = '+919849257983'
clean_num = '9849257983'

print('='*70)
print('CLOUD-STRIKE: Shadow Backup & Cloud Leak Detection')
print('='*70)
print(f'Target: {phone}')
print('Searching for cloud exposures and shadow backups...')
print()

cloud_dossier = {
    'phone_number': phone,
    'associated_emails': [],
    'cloud_exposures': [],
    'data_breaches': [],
    'shadow_backups': [],
}

print('[1/4] Searching data breaches for phone-email associations...')
print(f'  Target: {phone}')

leak_dorks = [
    f'{clean_num} (@ OR gmail OR yahoo) site:pastebin.com OR site:github.com',
    f'site:hudson.rock OR site:leak-check.net {phone}',
    f'{phone} (email OR mail OR @) filetype:csv OR filetype:json',
]

for i, dork in enumerate(leak_dorks, 1):
    print(f'  Leak dork {i}: {dork[:40]}...')
    try:
        google_url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=2'
        resp = requests.get(google_url, headers={'User-Agent': 'Mozilla/5.0'}, timeout=10)
        if resp.status_code == 200:
            emails = re.findall(r'[\w\.-]+@[\w\.-]+\.\w+', resp.text[:5000])
            if emails:
                unique_emails = list(set(emails))
                print(f'    Found {len(unique_emails[:3])} potential email associations')
                for email in unique_emails[:3]:
                    print(f'      - {email}')
                    cloud_dossier['associated_emails'].append(email)
    except Exception as e:
        print(f'    Error: {str(e)[:30]}')
    time.sleep(2)

print()
print('[2/4] Searching cloud storage for exposures...')
cloud_dorks = [
    f'filetype:json OR filetype:xml {clean_num}',
    f'site:dropbox.com OR site:drive.google.com {phone}',
]

for i, dork in enumerate(cloud_dorks, 1):
    print(f'  Cloud dork {i}')
    try:
        google_url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=2'
        resp = requests.get(google_url, headers={'User-Agent': 'Mozilla/5.0'}, timeout=10)
        cloud_urls = re.findall(r'https://[\w\-\.]+/[\w\-\.]+', resp.text[:3000])
        if cloud_urls and len(set(cloud_urls)) > 0:
            print(f'    Found {len(set(cloud_urls))} potential cloud URLs')
            for url in set(cloud_urls)[:2]:
                print(f'      - {url[:80]}')
                cloud_dossier['cloud_exposures'].append(url[:80])
    except Exception as e:
        print(f'    Error: {str(e)[:30]}')
    time.sleep(2)

print()
print('[3/4] Checking data breach databases...')
breaches = ['Yahoo', 'Adobe', 'LinkedIn', 'Dropbox']
for b in breaches:
    print(f'  Checking {b}... Status: Email association unknown (no public API)')
    cloud_dossier['data_breaches'].append({'breach': b, 'status': 'Unknown'})

print()
print('[4/4] Scanning for shadow backups...')
archive_dorks = [
    f'sites:web.archive.org {phone}',
    f'(backup OR archive) filetype:zip {phone}',
]

for i, dork in enumerate(archive_dorks, 1):
    print(f'  Archive dork {i}')
    try:
        google_url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=2'
        resp = requests.get(google_url, headers={'User-Agent': 'Mozilla/5.0'}, timeout=10)
        archive_urls = re.findall(r'https://web\.archive\.org/[^\s]+', resp.text[:3000])
        if archive_urls:
            print(f'    Found {len(archive_urls)} archived references')
            cloud_dossier['shadow_backups'].extend(archive_urls[:2])
    except Exception as e:
        print(f'    Error: {str(e)[:30]}')
    time.sleep(2)

print()
print('='*70)
print('CLOUD-STRIKE REPORT')
print('='*70)
print(f"Total Exposures: {len(cloud_dossier['associated_emails']) + len(cloud_dossier['cloud_exposures']) + len(cloud_dossier['shadow_backups'])}")
print(f"Email Associations: {len(cloud_dossier['associated_emails'])}")
print(f"Cloud Exposures: {len(cloud_dossier['cloud_exposures'])}")
print(f"Shadow Backups: {len(cloud_dossier['shadow_backups'])}")
print()

if (len(cloud_dossier['associated_emails']) == 0 and 
    len(cloud_dossier['cloud_exposures']) == 0 and 
    len(cloud_dossier['shadow_backups']) == 0):
    print('PRIVACY VERDICT: PROTECTED - No public cloud residue detected')
else:
    print('PRIVACY VERDICT: VULNERABLE - Cloud exposure residues detected')

print()
print('Saving results to result.txt')

# Save results to result.txt
with open('result.txt', 'w') as f:
    f.write(json.dumps(cloud_dossier, indent=2))
print(f'Cloud Strike results saved to result.txt')
