#!/usr/bin/env python3
import sys
import json
from datetime import datetime

# Add scripts directory to path
sys.path.insert(0, '.')

try:
    from scripts.cve_2026_41940_scan import scan_cve_2026_41940
except ImportError:
    # Fallback: try without scripts/ prefix
    from cve_2026_41940_scan import scan_cve_2026_41940

targets = [
    'whm.logitech.com',
    'whm.emerson.com'
]

print('Starting CVE-2026-41940 scan...')

for domain in targets:
    result = scan_cve_2026_41940(domain)
    
    if 'logitech' in domain:
        filename = 'security_scan_logitech.json'
    elif 'emerson' in domain:
        filename = 'security_scan_emerson.json'
    else:
        filename = f'security_scan_{domain}.json'
    
    with open(filename, 'w') as f:
        json.dump(result, f, indent=2)
    
    print(f'✓ Scan complete: {filename}')
    print(f'  Target: {domain}')
    print(f'  Status: {result["summary"]["security_status"]}')
    
    if result['summary']['security_status'] == 'VULNERABLE':
        print(f'  ⚠️ CVE-2026-41940 CONFIRMED: {domain}')

print('\nAll scans complete.')
