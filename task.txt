#!/usr/bin/env python3
import subprocess
import urllib.request
import time
import ssl

print('=== GHOST PROTOCOL: INITIATING ===')

def run_cmd(cmd):
    return subprocess.run(cmd, shell=True, capture_output=True, text=True)

# Phase 1: Ensure Tor is installed and running
print('Installing/Verifying Tor...')
run_cmd('sudo apt update && sudo apt install -y tor')
run_cmd('sudo systemctl start tor')
time.sleep(10) # Wait for Tor to initialize circuit

# Phase 2: Verify Ghost IP
print('Verifying Ghost Exit IP...')
proxy_support = urllib.request.ProxyHandler({'socks5h': '127.0.0.1:9050'})
opener = urllib.request.build_opener(proxy_support)
urllib.request.install_opener(opener)

try:
    with urllib.request.urlopen('http://icanhazip.com', timeout=60) as resp:
        exit_ip = resp.read().decode().strip()
        print(f'Exit IP (via Tor): {exit_ip}')
        
    print('=== GHOST PROTOCOL: HANDSHAKE TEST ===')
    # Use the same dummy test payload but through Tor
    url = 'https://results.manabadi.co.in/ts-10th-class-results-2026-regular-123456'
    headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)'}
    context = ssl._create_unverified_context()
    
    # We already have opener installed for global requests
    req = urllib.request.Request(url, headers=headers)
    with urllib.request.urlopen(req, context=context, timeout=60) as resp:
        print(f'Ghost Status: {resp.getcode()}')
        print('Status: Ghost Success/Verified')
    
    print('=== STATUS 200 ===')
except Exception as e:
    print(f'Ghost Error: {e}')
    print('=== STATUS 503 ===')
