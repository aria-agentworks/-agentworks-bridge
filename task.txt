#!/usr/bin/env python3
"""
Maya Growth - Mission 01
Chrome CDP Zero-Dependency Test
Target: http://4.213.114.95:9222 (localhost for self-hosted runner)
"""

import urllib.request
import json

print("=== MAYA GROWTH: MISSION 01 (ZERO-DEPENDENCY) ===")
print("Connecting to Chrome at http://4.213.114.95:9222 (localhost on Azure VM)")
print("\n--- Chrome CDP Discovery ---")

try:
    # Use only standard library - urllib.request
    response = urllib.request.urlopen("http://4.213.114.95:9222/json/version", timeout=5)
    version_info = response.read()
    data = json.loads(version_info)
    
    # Extract relevant info
    chrome_version = data.get("Browser", "N/A")
    web_socket_url = data.get("webkit DevTools Protocol URL", "N/A") or data.get("webSocketDebuggerUrl", "N/A")
    
    print(f"Browser: {chrome_version}")
    print(f"WebSocket URL: {web_socket_url}")
    print("\n=== STATUS 200 ===")
    print("Chrome connection successful!")
    print("=== Mission Success! ===")
    
except urllib.error.HTTPError as e:
    print(f"HTTP Error: {e.code} - {e.reason}")
    print("=== STATUS 503 ===")
    print("Chrome not responding at 4.213.114.95:9222")
except urllib.error.URLError as e:
    print(f"URL Error: {e.reason}")
    print("=== STATUS 503 ===")
    print("Cannot reach 4.213.114.95:9222")
except Exception as e:
    print(f"Error: {e}")
    print("=== STATUS 503 ===")
    print("Chrome not accessible")

print("=== END MISSION ===")
