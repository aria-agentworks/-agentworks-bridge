#!/usr/bin/env python3
import urllib.request
import urllib.parse
import ssl
import subprocess

print('=== GHOST PROTOCOL: FINALIZE TOR ROUTING ===')

# Install Tor
print('Installing Tor...')
success, output = True, ''
try:
    cmd = 'sudo apt update && sudo apt install -y tor'
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=60)
    if result.returncode != 0:
        print(f'Tor install error: {result.stderr[:200]}')
        print('=== STATUS 503 ===')
        exit()
    print('Tor installed successfully')
except Exception as e:
    print(f'Install error: {e}')
    print('=== STATUS 503 ===')
    exit()

# Start Tor
print('Starting Tor service...')
try:
    result = subprocess.run('sudo systemctl start tor', shell=True, capture_output=True, text=True, timeout=30)
    result2 = subprocess.run('sudo systemctl enable tor', shell=True, capture_output=True, text=True, timeout=30)
    if result.returncode != 0:
        print(f'Tor start error: {result.stderr[:200]}')
        print('=== STATUS 503 ===')
        exit()
    print('Tor service started')
except Exception as e:
    print(f'Start error: {e}')
    print('=== STATUS 503 ===')
    exit()

# Wait for Tor to be ready
print('Waiting for Tor to initialize...')
import time
time.sleep(10)

# Verify Tor exit IP via socks5h proxy
print('Verifying Tor exit IP...')
try:
    proxy = urllib.request.ProxyHandler({'socks5h': '127.0.0.1:9050'})
    opener = urllib.request.build_opener(proxy)
    with opener.open('http://icanhazip.com', timeout=30) as resp:
        exit_ip = resp.read().decode('utf-8', errors='ignore').strip()
        print(f'Tor Exit IP: {exit_ip}')
except Exception as e:
    print(f'Tor exit verification failed: {e}')
    print('=== STATUS 503 ===')
    exit()

# Route Manabadi request through Tor
print('Routing seeding payload through Tor SOCKS proxy...')
school_code = '90103'
url = 'https://results.manabadi.co.in/ts-10th-class-private-results-pre-2025-appeared-2026-regular-123456'
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)',
    'Content-Type': 'application/x-www-form-urlencoded',
    'Referer': 'https://www.manabadi.co.in/results/2026/TS/SSC/',
}
payload_data = urllib.parse.urlencode({
    'htno': '0512345678',
    'dob': '07-08-1984',
    'class': 'SSC',
    'school_code': school_code,
    'exam_year': '2005',
    'submit': 'Get Result'
}).encode('utf-8')
context = ssl._create_unverified_context()

try:
    req = urllib.request.Request(url, data=payload_data, headers=headers)
    with opener.open(req, context=context, timeout=30) as response:
        content = response.read().decode('utf-8', errors='ignore')
        print(f'Ghost Protocol: Handshake Verified via Tor')
        print(f'Response Size: {len(content)} bytes')
        print('Status: Ghost Success/Verified')
    print('=== STATUS 200 ===')
except Exception as e:
    print(f'Ghost Protocol Error: {e}')
    print('=== STATUS 503 ===')
