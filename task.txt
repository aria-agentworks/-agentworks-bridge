#!/usr/bin/env python3
import subprocess

# Navigate to GitHub and capture page info to verify the browser is alive
browser_code = "goto_url('https://github.com'); sleep(2); print(page_info())"
cmd = f"browser-harness -c \"{browser_code}\""

print(f"Running Remote Browser Test: {cmd}")
try:
    res = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    print(f"Output: {res.stdout}")
    print(f"Error: {res.stderr}")
except Exception as e:
    print(f"Failed to execute browser-harness: {e}")
