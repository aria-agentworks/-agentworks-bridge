#!/usr/bin/env python3
import os

# Install dependencies in the bridge environment
os.system('pip3 install browser-harness cdp-use --break-system-packages')

from browser_harness.run import goto_url, page_info, capture_screenshot, set_cdp_url

# Point the harness to the Azure VM's remote Chrome
set_cdp_url('http://4.213.114.95:9222')

print("Connecting to Remote Browser on Azure VM...")
try:
    goto_url('https://www.instagram.com/natural_beauty_shoutout/')
    import time
    time.sleep(5)
    print(f"Page Info: {page_info()}")
except Exception as e:
    print(f"Error connecting to remote browser: {e}")
