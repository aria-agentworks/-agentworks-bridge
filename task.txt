#!/bin/bash
echo "=== Robust Remote Browser Setup ==="

# 1. Kill any zombie Chrome processes
sudo pkill -f chrome || true

# 2. Ensure Chrome and dependencies are present
sudo apt-get update && sudo apt-get install -y google-chrome-stable chromium-browser

# 3. Re-install browser-harness to a specific location
pip3 install browser-harness cdp-use --break-system-packages

# 4. Find the absolute path of browser-harness
BH_PATH=$(which browser-harness || find /usr/local/bin /home/azureuser/.local/bin -name browser-harness | head -n 1)
echo "Found browser-harness at: $BH_PATH"

# 5. Start Chrome with remote debugging (background, nohup)
nohup google-chrome-stable --headless --remote-debugging-port=9222 --user-data-dir=/tmp/chrome-profile --no-sandbox --disable-dev-shm-usage > /tmp/chrome.log 2>&1 &

# 6. Wait for Chrome to warm up
echo "Waiting for Chrome to start..."
for i in {1..10}; do
  if curl -s http://localhost:9222/json/version > /dev/null; then
    echo "✅ Chrome is ALIVE on port 9222"
    break
  fi
  sleep 2
done

# 7. Run the Navigation Test using the absolute path
if [ -n "$BH_PATH" ]; then
  echo "Running GitHub test with $BH_PATH..."
  $BH_PATH -c "goto_url('https://github.com'); sleep(3); print(page_info())"
else
  echo "❌ browser-harness still not found. Trying via python module..."
  python3 -m browser_harness.run -c "goto_url('https://github.com'); sleep(3); print(page_info())"
fi

echo "=== Setup Complete ==="
