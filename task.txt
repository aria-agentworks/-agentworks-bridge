#!/usr/bin/env python3
import subprocess
import urllib.request
import time
import ssl
import sys

print('=== GHOST PROTOCOL: INITIATING RECOVERY ===')

def run_cmd(cmd):
    return subprocess.run(cmd, shell=True, capture_output=True, text=True)

# Phase 1: Force Restart Tor and verify port
print('Restarting Tor service...')
run_cmd('sudo systemctl restart tor')

# Loop to wait for Tor SOCKS port 9050
for i in range(10):
    res = run_cmd('ss -tlnp | grep 9050')
    if '9050' in res.stdout:
        print('Tor SOCKS port 9050 is listening.')
        break
    print('Waiting for Tor port...')
    time.sleep(5)

# Phase 2: Set up Opener with SSL Context and SOCKS5h
context = ssl._create_unverified_context()
proxy_handler = urllib.request.ProxyHandler({'http': 'socks5h://127.0.0.1:9050', 'https': 'socks5h://127.0.0.1:9050'})
https_handler = urllib.request.HTTPSHandler(context=context)
opener = urllib.request.build_opener(proxy_handler, https_handler)

# Phase 3: Verify Exit IP
print('Verifying Ghost Exit IP...')
azure_ip = '4.213.114.95'
exit_ip = None

for i in range(12): # Wait up to 60s for circuit
    try:
        req = urllib.request.Request('http://icanhazip.com', headers={'User-Agent': 'curl/7.68.0'})
        with opener.open(req, timeout=10) as resp:
            exit_ip = resp.read().decode().strip()
            print(f'Current Exit IP: {exit_ip}')
            if exit_ip != azure_ip:
                print('Ghost Protocol Verified: Exit node is non-attributable.')
                break
    except Exception as e:
        print(f'Waiting for circuit... ({e})')
    time.sleep(5)

if not exit_ip or exit_ip == azure_ip:
    print('FATAL: Ghost Protocol failed to establish non-attributable circuit.')
    print('=== STATUS 503 ===')
    sys.exit(1)

# Phase 4: Fire Ghosted Handshake
try:
    print('=== GHOST PROTOCOL: HANDSHAKE TEST ===')
    url = 'https://results.manabadi.co.in/ts-10th-class-results-2026-regular-123456'
    headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)'}
    
    req = urllib.request.Request(url, headers=headers)
    with opener.open(req, timeout=30) as resp:
        print(f'Ghost Status: {resp.getcode()}')
        print('Status: Ghost Success/Verified')
    
    print('=== STATUS 200 ===')
except Exception as e:
    print(f'Ghost Handshake Error: {e}')
    print('=== STATUS 503 ===')
