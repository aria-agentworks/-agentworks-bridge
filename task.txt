    f'site:web.archive.org "{handle}" OR "{clean}"',
    f'site:web.archive.org "{name}" OR "Himabindu Atluri"',
]

    print(f"  Searching: {dork[:50]}...")
    url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=2'
    r = requests.get(url, headers={'User-Agent': 'Mozilla'}, timeout=10)
    if r.status_code == 200:
        links = re.findall(r'(https://[^\s]+(?:web\.archive\.org|wayback\.archive\.org)[^\s]*)', r.text[:5000])
        for l in list(set(links))[:3]:
            if len(l) < 150:
                print(f"    Found archive: {l[:120]}")
                report['footprints'].append({'type': 'archive', 'url': l[:120], 'name': name})
    time.sleep(2)

print("\n[2/5] Searching Personal Blogs & Portfolio Sites")
blog_dorks = [
    f'site:wordpress.com "{name}" OR "Himabindu"',
    f'site:medium.com "{name}" OR "{handle}"',
    f'site:blogger.com "{name}"',
    f'site:about.me "{name}" OR "{handle}"',
]

for dork in blog_dorks:

    url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=1'
    r = requests.get(url, headers={'User-Agent': 'Mozilla'}, timeout=10)
    if r.status_code == 200:

    if r.status_code == 200:
        posts = re.findall(r'(https://www\.reddit\.com/r/[^\s]+/[^\s]+)', r.text[:5000])
        for post in list(set(posts))[:3]:
            print(f"    Found forum post: {post[:130]}")
            report['forums'].append({'url': post[:130], 'name': name})
            report['footprints'].append({'type': 'forum', 'url': post[:130], 'name': name})

print("\n[4/5] Searching Old Profile Sites (Geocities, etc.)")

    f'site:xanga.com "{name}"',

    r = requests.get(url, headers={'User-Agent': 'Mozilla'}, timeout=10)
    if r.status_code == 200:
        old_sites = re.findall(r'(https://[^\s]+(?:geocities\.com|xanga\.com|livejournal\.com|bizrate\.com)[^\s]*)', r.text[:5000])
        for site in list(set(old_sites))[:2]:
            if len(site) < 150:
                print(f"    Found old site: {site[:120]}")
                report['archives'].append({'url': site[:120], 'type': 'old'})
    time.sleep(2)

print("\n[5/5] Searching for Email Links in Historical Data")
email_dorks = [
    f'site:pastebin.com "{name}" +email',
    f'site:hudson.rock "{clean}" +email',
    f'site:leakcheck.net "{handle}"',
]

for dork in email_dorks:
    print(f"  Searching: {dork[:40]}...")
    url = f'https://www.google.com/search?q={urllib.parse.quote(dork)}&num=1'
    r = requests.get(url, headers={'User-Agent': 'Mozilla'}, timeout=10)
    if r.status_code == 200:
        emails = re.findall(r'[\w\.-]+@[\w\.-]+\.\w+', r.text[:5000])
        for e in list(set(emails)):
            if e.count('@') == 1 and '.' in e:
                print(f"    Found email: {e[:60]}")
                report['vulnerabilities'].append({'type': 'email', 'value': e, 'context': f"Linked to {name}"})
    time.sleep(2)
