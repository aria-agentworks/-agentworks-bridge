=== VM IDENTITY CHECK ===
#!/bin/bash

echo "Hostname: $(hostname)"
echo "Current User: $USER"
echo "IP Address: $(hostname -I 2>/dev/null || ifconfig -a 2>/dev/null | grep 'inet ' | grep -v 127.0.0.1 | awk '{print $2}' | tr '\n' ' ' || ip addr show | grep 'inet ' | grep -v 'inet6' | awk '{print $2}' | tr '\n' ' ' || echo 'N/A')"

echo "Sudo Test:"
sudo -v >/dev/null 2>&1 && echo "root" || echo "no root"

echo "Docker Test:"
docker --version 2>/dev/null || echo "Docker not installed"
echo "=== END IDENTITY CHECK ==="
