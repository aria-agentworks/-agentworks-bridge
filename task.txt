#!/bin/bash
echo "=== Maya Growth: Mission 01 (Remote Browser) ==="

# 1. Install dependencies
pip3 install browser-harness cdp-use --break-system-packages --quiet

# 2. Find browser-harness binary
BH_BIN=$(find /usr/local/bin /usr/bin /home/azureuser/.local/bin -name browser-harness -type f 2>/dev/null | head -1)

if [ -z "$BH_BIN" ]; then
    echo "ERROR: browser-harness not found"
    exit 1
fi

echo "Found browser-harness at: $BH_BIN"

# 3. Run browser-harness CLI with Python code passed as argument
eval $BH_BIN -c '
import time
import json
from browser_harness import Harness

try:
    h = Harness()
    h.set_cdp_url("http://4.213.114.95:9222")
    print("Connecting to Remote Browser...")
    h.goto_url("https://www.instagram.com/natural_beauty_shoutout/")
    time.sleep(5)
    info = h.page_info()
    print("=== page_info() ===")
    print(json.dumps(info, indent=4))
    if info.get("status") == 200:
        print("=== STATUS 200 ===")
        print("Mission Success!")
except Exception as e:
    print(f"Error: {e}")
'

echo "=== Mission Test Complete ==="
