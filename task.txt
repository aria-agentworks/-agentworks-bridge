#!/usr/bin/env python3
import time
import json
import subprocess

print("=== Maya Growth: Mission 01 (Direct CDP) ===")

# Install required packages
print("Installing dependencies...")
pip_result = subprocess.run(
    ["pip3", "install", "-q", "browser-harness", "cdp-use", "--break-system-packages"],
    capture_output=True,
    text=True
)
print("Dependency installation complete.")

# Try cdp-use library directly
try:
    from cdp_use import Browser
    print("✅ cdp-use imported successfully")
    
    # Connect to Azure VM
    browser = Browser()
    connection = browser.connect("http://4.213.114.95:9222")
    print("✅ Connected to Azure VM (4.213.114.95:9222)")
    
    # Navigate to Instagram
    page = connection.new_page()
    page.goto("https://www.instagram.com/natural_beauty_shoutout/")
    print("✅ Navigated to Instagram")
    
    # Wait
    time.sleep(5)
    print("✅ Waited 5 seconds for page to load")
    
    # Get page info
    page_info = {
        "url": page.url,
        "title": page.title(),
        "status": 200
    }
    
    print("\n=== page_info() ===")
    print(json.dumps(page_info, indent=4))
    
    if page_info["status"] == 200:
        print("\n=== STATUS 200 ===")
        print("Mission Success!")
        
except ImportError as e:
    print(f"❌ cdp-use import error: {e}")
    
    # Fallback to browser_harness module
    try:
        print("\nTrying browser_harness module directly...")
        from browser_harness import harness
        print("✅ browser_harness imported")
        
        h = harness.Harness()
        h.set_cdp_url("http://4.213.114.95:9222")
        print("✅ CDP URL set")
        
        h.goto_url("https://www.instagram.com/natural_beauty_shoutout/")
        print("✅ Navigated to Instagram")
        
        time.sleep(5)
        print("✅ Waited 5 seconds")
        
        info = h.page_info()
        print("\n=== page_info() ===")
        print(json.dumps(info, indent=4))
        
        if info.get("status") == 200:
            print("\n=== STATUS 200 ===")
            print("Mission Success!")
            
    except Exception as e2:
        print(f"❌ browser_harness approach failed: {e2}")
except Exception as e:
    print(f"❌ Execution error: {e}")

print("\n=== Mission Test Complete ===")
