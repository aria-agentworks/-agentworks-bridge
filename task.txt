#!/usr/bin/env python3
import urllib.request
import urllib.parse
import json

print('=== Shadow Architect: TS Board Handshake Test ===')
# Testing the 'historical' mirror while 2026 load is peak
url = 'https://results.manabadi.co.in/ts-10th-class-private-results-pre-2025-appeared-2026-regular-123456' # Example endpoint from research
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)',
    'Content-Type': 'application/x-www-form-urlencoded',
    'Referer': 'https://www.manabadi.co.in/results/2026/TS/SSC/',
    'Origin': 'https://www.manabadi.co.in'
}
# Using a dummy hall ticket for a historical year (e.g. 2005)
data = urllib.parse.urlencode({
    'htno': '0512345678', # Dummy historical HT
    'dob': '15-05-1989',
    'class': 'SSC',
    'exam_year': '2005',
    'submit': 'Get Result'
}).encode('utf-8')

try:
    print(f'Sending POST to {url}...')
    req = urllib.request.Request(url, data=data, headers=headers)
    with urllib.request.urlopen(req, timeout=30) as response:
        status = response.getcode()
        content = response.read().decode('utf-8', errors='ignore')
        print(f'=== Handshake Status: {status} ===')
        print(f'Response Size: {len(content)} bytes')
        if 'Result Not Found' in content or 'Invalid' in content:
            print('Handshake established: Legacy DB queried successfully.')
        else:
            print('Unexpected response structure.')
    print('=== STATUS 200 ===')
except Exception as e:
    print(f'Error: {e}')
    print('=== STATUS 503 ===')
