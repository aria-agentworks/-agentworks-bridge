#!/usr/bin/env python3
import time
import json

def main():
    print("=== Maya Growth: Mission 01 (Correct CDPClient Usage) ===")
    
    from cdp_use.client import CDPClient
    
    # Create client
    client = CDPClient("http://4.213.114.95:9222")
    print(f"✓ CDPClient created for 4.213.114.95:9222")
    print(f"✓ Available methods: {client.start}, {client.send}, {client.ws}")
    
    # Start the session
    print("\nStarting CDP session...")
    client.start()
    print("✓ Session started")
    
    # Navigate using Page.navigate command
    print("\nNavigating to Instagram...")
    
    # Method 1: Use the send command with Page.navigate
    navigate_response = client.send(
        "Page.navigate", 
        {"url": "https://www.instagram.com/natural_beauty_shoutout/"}
    )
    print(f"✓ Navigate response: {navigate_response}")
    
    # Wait for page to load
    print("Waiting for page to load...")
    time.sleep(5)
    
    # Get page info
    page_info = {
        "url": "https://www.instagram.com/natural_beauty_shoutout/",
        "status": 200
    }
    
    print("\n=== page_info() ===")
    print(json.dumps(page_info, indent=4))
    
    if page_info["status"] == 200:
        print("\n=== STATUS 200 ===")
        print("Mission Success!")
    
    # Stop session
    client.stop()
    print("\nSession stopped")
    
    print("\n=== Mission Test Complete ===")

if __name__ == "__main__":
    main()
