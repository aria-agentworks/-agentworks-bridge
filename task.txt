#!/usr/bin/env python3
import subprocess
import os
import sys

print("=== Maya Growth: Mission 01 (Module Discovery) ===")

# Install packages
print("Installing...")
subprocess.run(["pip3", "install", "-q", "browser-harness", "cdp-use", "--break-system-packages"], check=False)

# Find module locations
print("\n1. Finding module locations...")
result = subprocess.run(
    ["python3", "-c", "import browser_harness; import os; print(os.path.dirname(browser_harness.__file__))"],
    capture_output=True, text=True
)
if result.returncode == 0:
    BH_PATH = result.stdout.strip()
    print(f"   browser_harness: {BH_PATH}")
else:
    print("   browser_harness not found")

result = subprocess.run(
    ["python3", "-c", "import cdp_use; import os; print(os.path.dirname(cdp_use.__file__))"],
    capture_output=True, text=True
)
if result.returncode == 0:
    CDP_PATH = result.stdout.strip()
    print(f"   cdp_use: {CDP_PATH}")

# List files in browser_harness
print("\n2. Inspecting browser_harness module...")
if 'BH_PATH' in locals():
    try:
        result = subprocess.run(["ls", "-la", BH_PATH], capture_output=True, text=True)
        print(f"   Files in browser_harness:\n{result.stdout}")
        
        # Search for .py files
        py_files = subprocess.run(["ls", f"{BH_PATH}/*.py"], capture_output=True, text=True)
        if py_files.returncode == 0:
            print(f"   Python files:\n{py_files.stdout}")
            
            # Read the first one
            if py_files.stdout.strip():
                first_file = py_files.stdout.strip().split("\n")[0]
                print(f"\n   Contents of {first_file}:")
                with open(first_file, "r") as f:
                    content = f.read(2000)
                    print(content[:2000])
    except Exception as e:
        print(f"   Error: {e}")

# Search in cdp_use
print("\n3. Inspecting cdp_use module...")
if 'CDP_PATH' in locals():
    try:
        result = subprocess.run(["ls", "-la", CDP_PATH], capture_output=True, text=True)
        print(f"   Files in cdp_use:\n{result.stdout}")
        
        py_files = subprocess.run(["ls", f"{CDP_PATH}/*.py"], capture_output=True, text=True)
        if py_files.returncode == 0:
            print(f"   Python files:\n{py_files.stdout}")
            
            # Read the main file
            if py_files.stdout.strip():
                first_file = py_files.stdout.strip().split("\n")[0]
                print(f"\n   Contents of {first_file}:")
                with open(first_file, "r") as f:
                    content = f.read(2000)
                    print(content[:2000])
    except Exception as e:
        print(f"   Error: {e}")

print("\n=== Discovery Complete ===")
