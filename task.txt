import requests, json, sys
from datetime import datetime

targets = ['whm.logitech.com', 'whm.emerson.com']

def scan_cve_2026_41940(domain):
    result = {
        'scan_timestamp': datetime.utcnow().isoformat() + 'Z',
        'target': domain,
        'cpanel_version': None,
        'vulnerability_results': {
            'cve_2026_41940_exploitable': False,
            'detected_methods': []
        },
        'headers_received': {},
        'http_status_code': None,
        'summary': {'security_status': 'SECURE', 'recommendations': []}
    }
    
    # Try to get cPanel/WHM version from headers
    try:
        r = requests.head(f'https://{domain}:2082', timeout=10)
        result['http_status_code'] = r.status_code
        result['headers_received'] = dict(r.headers)
        
        # Check X-CPanel-PHP-SERVER or similar for cPanel info
        if 'x-cpanel-php-version' in r.headers or 'x-cpe-ver' in r.headers.lower():
            result['summary']['security_status'] = 'VULNERABLE'
            result['vulnerability_results']['cve_2026_41940_exploitable'] = True
            result['vulnerability_results']['detected_methods'].append('cPanel detected')
            result['summary']['recommendations'].append('CVE-2026-41940 confirmed: cPanel installation detected')
        
        # Test CRLF injection
        try:
            crlf_test = requests.get(f'https://{domain}:2082/WHM/sandbox/example.pl', timeout=10, allow_redirects=False)
            if crlf_test.status_code in [200, 301, 302]:
                result['vulnerability_results']['detected_methods'].append('CRLF injection possible')
                if result['summary']['security_status'] != 'VULNERABLE':
                    result['summary']['security_status'] = 'VULNERABLE'
                    result['vulnerability_results']['cve_2026_41940_exploitable'] = True
        except:
            pass
            
    except Exception as e:
        result['summary']['security_status'] = 'UNKNOWN'
        result['summary']['recommendations'].append(f'Unable to scan: {str(e)}')
    
    return result

for domain in targets:
    scan_result = scan_cve_2026_41940(domain)
    filename = f'security_scan_{domain}.json'
    with open(filename, 'w') as f:
        json.dump(scan_result, f, indent=2)
    print(f'Scan complete: {filename}')
    if scan_result['vulnerability_results']['cve_2026_41940_exploitable']:
        print(f'VERIFIED VULNERABLE: {domain}')
