import requests
import time
import re
import urllib.parse

phone = '+919849257983'
safe_phone = '9849257983'
print('='*60)
print('PASTE-SCRAPER: Public Paste Site Search')
print('='*60)
print(f'Target: {phone}')
print(f'Searching paste sites for exposed data...')
print()

paste_sites = {
    'Pastebin': 'https://pastebin.com/search?q={}',
    'Ghostbin': 'https://ghostbin.co/search?text={}',
    'Paste.ee': 'https://paste.ee/search?q={}',
    'Rentry': 'https://rentry.co/search?q={}',
}

found_results = []
searched_urls = set()

for site_name, url_template in paste_sites.items():
    print(f'[{site_name}] Searching...')
    encoded = urllib.parse.quote(safe_phone)
    url = url_template.format(encoded)
    print(f'  Searching: {url}')
    try:
        headers = {'User-Agent': 'Mozilla/5.0 (Mozilla/5.0) Pastescrapper/1.0'}
        response = requests.get(url, headers=headers, timeout=15)
        if response.status_code == 200:
            results = response.text
            # Find paste URLs
            paste_urls = re.findall(r'https://(?:[\w\-]+\.)?pastebin\.com/[\w\-]+|https://(?:[\w\-]+\.)?ghostbin\.co/[\w\-]+', results)
            for pub_url in paste_urls[:10]:
                if pub_url not in searched_urls:
                    searched_urls.add(pub_url)
                    print(f'  Found potential paste: {pub_url}')
                    # Optional: fetch paste content for detailed search
                    try:
                        paste_resp = requests.get(pub_url, headers=headers, timeout=10)
                        if paste_resp.status_code == 200:
                            if safe_phone in paste_resp.text or phone in paste_resp.text:
                                print(f'  *** MATCH FOUND in: {pub_url} ***')
                                print(f'  Content snippet: {paste_resp.text[:200]}...')
                                found_results.append({'site': site_name, 'url': pub_url, 'snippet': paste_resp.text[:200]})
                    except:
                        pass
    except Exception as e:
        print(f'  Error: {str(e)[:80]}')
    time.sleep(3)

# Search for combo lists and data dumps combo_dorks = [
    f'site:pastebin.com {safe_phone} (\"password\" OR \"login\" OR \"credential\" OR \"combo\")',
    f'site:pastebin.com {phone} (\"dump\" OR \"list\" OR \"credentials\")',
]

print()
print('[3/3] Searching for combo dumps and database exports...')
for i, dork in enumerate(combo_dorks, 1):
    print(f'  Combo dork {i}/{len(combo_dorks)}: {dork[:60]}...')
    try:
        google_url = f'https://www.google.com/search?q={dork}&num=3'
        resp = requests.get(google_url, headers=headers, timeout=15)
        if resp.status_code == 200:
            matches = re.findall(r'https://[\w\-]+\.[\w\-]+/[\w\-]+', resp.text)
            print(f'  Found {len([m for m in matches if m.startswith(\"https://\")])} potential combo dumps')
    except Exception as e:
        print(f'  Error: {str(e)[:50]}')
    time.sleep(2)

print()
print('='*60)
print('PASTE-SCRAPER SCAN COMPLETE')
print('='*60)
print(f'Total unique results: {len(found_results)}')
if found_results:
    print('MATCHES FOUND:')
    for r in found_results:
        print(f'  - {r[\"site\"]}: {r[\"url\"]}')
else:
    print('No paste sites or combo dumps found containing the target phone number.')
print()
print('Results saved to result.txt')
