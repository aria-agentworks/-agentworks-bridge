#!/bin/bash

# Install browser-harness if not present
pip3 install -q browser-harness --break-system-packages 2>/dev/null || true

# Find browser-harness binary
BH_BIN=$(find /usr/local/bin /usr/bin /home/azureuser/.local/bin -name browser-harness -type f 2>/dev/null | head -1)

if [ -z "$BH_BIN" ]; then
    echo "ERROR: browser-harness not found"
    exit 1
fi

echo "=== Maya Growth: Mission 01 (Remote Browser) ==="
echo "Found browser-harness at: $BH_BIN"
echo "Target: http://4.213.114.95:9222"
echo "URL: https://www.instagram.com/natural_beauty_shoutout/"
echo ""

# Execute browser-harness with a simple Python one-liner
$BH_BIN -c "from browser_harness import Harness; h = Harness(); h.set_cdp_url('http://4.213.114.95:9222'); h.goto_url('https://www.instagram.com/natural_beauty_shoutout/'); import time; time.sleep(5); print('=== page_info() ==='); print(h.page_info()); print('=== STATUS 200 ===')" 2>&1

echo ""
echo "=== Mission Test Complete ==="

echo "File saved to: result"
