#!/usr/bin/env python3

print("Testing Instagram Profile Navigation\n")

# Test 1: Navigate to Maya's official Instagram profile
print("Test 1: Navigating to https://www.instagram.com/officialmaya.ai/")
cmd1 = "browser-harness -c \"goto_url('https://www.instagram.com/officialmaya.ai/'); sleep(3); print(page_info())\""

try:
    print(f"Running: {cmd1}")
    res1 = subprocess.run(cmd1, shell=True, capture_output=True, text=True, timeout=60)
    print("=== Test 1 Result ===")
    print(res1.stdout)
    if res1.stderr:
        print("Errors:", res1.stderr)
except Exception as e:
    print(f"Test 1 failed: {e}")
