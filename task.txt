#!/usr/bin/env python3
import requests
import json

# Triggering Maya Dubai Post
image_url = "https://pub-85573262a9e846c5896860df66eb8299.r2.dev/img/gen_1777443799397_276427e8-525f-4232-8cf5-ea05bbd4df71.png"
caption = "Dubai nights. ✨ #maya #luxury #lifestyle"
webhook_url = "http://4.213.114.95:5678/webhook/maya"

payload = {
    "image_url": image_url,
    "caption": caption
}

print(f"Sending payload to {webhook_url}...")
try:
    response = requests.post(webhook_url, json=payload, headers={"Content-Type": "application/json"})
    print(f"Status Code: {response.status_code}")
    print(f"Response Body: {response.text}")
except Exception as e:
    print(f"Error: {e}")
