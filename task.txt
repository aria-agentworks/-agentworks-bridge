import urllib.request
import json
import time

print('=== Maya Growth: Mission 01 (Zero-Dependency) ===')
target_url = 'https://www.instagram.com/natural_beauty_shoutout/'
cdp_base = 'http://4.213.114.95:9222'

try:
    # Create new tab and navigate
    print(f'Creating new tab for: {target_url}')
    with urllib.request.urlopen(f'{cdp_base}/json/new?{target_url}') as response:
        data = json.loads(response.read().decode())
        print(f'Tab Created: {data["id"]}')
    
    # Wait for load
    time.sleep(5)
    
    # Verify tab
    with urllib.request.urlopen(f'{cdp_base}/json') as response:
        tabs = json.loads(response.read().decode())
        for tab in tabs:
            if target_url in tab.get('url', ''):
                print('=== page_info() ===')
                print(json.dumps({'url': tab['url'], 'title': tab['title'], 'status': 200}, indent=4))
                print('=== STATUS 200 ===')
                print('Mission Success!')
                break
except Exception as e:
    print(f'Error: {e}')

print('=== Mission Test Complete ===")
