#!/usr/bin/env python3
import subprocess
import urllib.request
import time
import ssl
import sys
import socket
import urllib.error

print('=== GHOST PROTOCOL: FINAL FIX ===')

def run_cmd(cmd):
    return subprocess.run(cmd, shell=True, capture_output=True, text=True)

# Phase 1: Ensure Tor and dependencies
print('Installing PySocks...')
run_cmd('pip3 install PySocks')
print('Restarting Tor...')
run_cmd('sudo systemctl restart tor')

# Wait for Tor port 9050
for _ in range(5):
    if '9050' in run_cmd('ss -tlnp | grep 9050').stdout: break
    time.sleep(5)

# Phase 2: Force SOCKS5 through PySocks
try:
    import socks
    socks.set_default_proxy(socks.SOCKS5, "127.0.0.1", 9050, rdns=True)
    socket.socket = socks.socksocket
    print('SOCKS5 Tunnel forced at socket level.')
except ImportError:
    print('PySocks not found.')

# Phase 3: Verify Anonymity
azure_ip = '4.213.114.95'
print(f'Verifying exit node. Azure IP: {azure_ip}')

try:
    context = ssl._create_unverified_context()
    req = urllib.request.Request('http://icanhazip.com', headers={'User-Agent': 'curl/7.68.0'})
    with urllib.request.urlopen(req, timeout=30) as resp:
        exit_ip = resp.read().decode().strip()
        print(f'Current Exit IP: {exit_ip}')
        if exit_ip == azure_ip:
            print('FATAL ERROR: IP LEAK DETECTED.')
            sys.exit(1)
        print('GHOST PROTOCOL VERIFIED: IP is non-attributable.')
except Exception as e:
    print(f'Tunnel verification failed: {e}')
    sys.exit(1)

# Phase 4: Execute Ghosted Handshake with retries
try:
    print('=== GHOST PROTOCOL: SECURE HANDSHAKE ===')
    url = 'https://results.manabadi.co.in/ts-10th-class-results-2026-regular-123456'
    context = ssl._create_unverified_context()
    
    max_retries = 3
    for attempt in range(max_retries):
        try:
            req = urllib.request.Request(url, headers={'User-Agent': 'Mozilla/5.0'})
            with urllib.request.urlopen(req, context=context, timeout=60) as resp:
                print(f'Handshake Status: {resp.getcode()}')
                print('Status: Ghost Success/Verified')
                print('=== STATUS 200 ===')
                break
        except urllib.error.URLError as e:
            if attempt < max_retries - 1:
                print(f'Timed out, retrying... ({attempt + 1}/{max_retries})')
                time.sleep(10)
            else:
                raise
    
except Exception as e:
    print(f'Handshake failed: {e}')
    print('=== STATUS 503 ===')
