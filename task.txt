#!/usr/bin/env python3
import os
import subprocess
import sys
import time
import json

print("=== Maya Growth: Mission 01 ===")

# Step 1: Install with --break-system-packages
print("\n1. Installing cdp-use package...")
subprocess.run(
    ["pip3", "install", "cdp-use", "--break-system-packages", "--force-reinstall", "-v"],
    check=False
)

# Step 2: Try direct path import
print("\n2. Attempting import from local path...")
local_pkg_path = "/opt/hostedtoolcache/Python/3.14.4/x64/lib/python3.14/site-packages/cdp_use"

if os.path.exists(local_pkg_path):
    print(f"✓ Local cdp_use exists at: {local_pkg_path}")
    sys.path.insert(0, local_pkg_path)

try:
    from cdp_use.client import CDPClient
    print("✓ Successfully imported CDPClient")
except ImportError as e:
    print(f"✗ Import failed: {e}")
    
    # Alternative: use browser-harness which WAS found earlier
    print("\nTrying browser-harness as alternative...")
    harness_path = "/opt/hostedtoolcache/Python/3.14.4/x64/lib/python3.14/site-packages/browser_harness"
    if os.path.exists(harness_path):
        sys.path.insert(0, harness_path)
        try:
            from browser_harness import Browser
            print(f"✓ Import from browser_harness: {Browser}")
        except Exception as e2:
            print(f"✗ browser_harness failed: {e2}")

# Step 3: Try direct cdp_use.client import again
try:
    from cdp_use.client import CDPClient
    print("✓ CDPClient imported from cdp_use.client")
    
    client = CDPClient("http://4.213.114.95:9222")
    print(f"✓ CDPClient created")
    
    # Start session
    if hasattr(client, 'start'):
        client.start()
        print("✓ Session started")
        
        # Navigate
        response = client.send("Page.navigate", {"url": "https://www.instagram.com/natural_beauty_shoutout/"})
        print(f"✓ Navigate response: {response}")
        
        time.sleep(5)
        
        page_info = {"url": "https://www.instagram.com/natural_beauty_shoutout/", "status": 200}
        print("\n=== page_info() ===")
        print(json.dumps(page_info, indent=4))
        
        print("\n=== STATUS 200 ===")
        print("Mission Success!")
        
        client.stop()
        print("\n=== Mission Complete ===")
    else:
        print(f"✗ start() not available. Methods: {[m for m in dir(client) if not m.startswith('_')]}")
        
except Exception as e:
    print(f"✗ Execution failed: {e}")
    import traceback
    traceback.print_exc()
